\par{Reduction contexts}

\begin{grammar}
  \bfseries
  <E> ::= [] \| \meta{E e}
  \alt \meta{E.w} \| \meta{E.w} or \meta{e} \| \meta{v.F} \| \meta{v.F} or \meta{e}
  \alt \{ \x/ = \e/; \ldots{}; \x/ = \e/; \E/ = \e/; \ldots{}; \e/ = \e/ \}
  \alt with \E/; \e/ | if \E/ then \e/ else \e/

  <F> ::= \E/.\a/ \| \v/.\meta{F}
\end{grammar}
